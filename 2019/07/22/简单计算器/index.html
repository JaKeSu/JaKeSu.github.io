简单计算器

发表于 2019-07-22 | 分类于 进军PAT
字数统计: 671 | 阅读时长 ≈ 3

题目描述

读入一个只包含 +, -, *, / 的非负整数计算表达式，计算该表达式的值。

输入

测试输入包含若干测试用例，每个测试用例占一行，每行不超过200个字符，整数和运算符之间用一个空格分隔。没有非法表达式。当一行中只有0时输入结束，相应的结果不要输出。

输出

对每个测试用例输出1行，即该表达式的值，精确到小数点后2位。

样例输入

30 / 90 - 26 + 97 - 5 - 6 - 13 / 88 * 6 + 51 / 29 + 79 * 87 + 57 * 920

样例输出

12178.21

#include<cstdio>
#include<iostream>
#include<stack>
#include<string>
#include<queue>
#include<map>
using namespace std;

struct node{//存储操作符或者数字
    double num;
    char op;
    bool flag;
};

string str;//记录读入的计算式
stack<node> s;
queue<node> q;
map<char,int>op;

void Change(){
    node temp;//接受操作数或者操作符
    for(int i=0;i<str.length();){
        if(str[i]>='0'&&str[i]<='9'){
            temp.flag=true;//读入的是操作数
            temp.num=str[i++]-'0';
            while(i<str.length()&&str[i]>='0'&&str[i]<='9'){//计算操作数
                temp.num=temp.num*10+(str[i]-'0');
                i++;
            }
            q.push(temp);//放入后缀表达式队列
        }else
        {
            temp.flag=false;
            while(!s.empty()&&op[str[i]]<=op[s.top().op]){//当读入的操作符的优先级小于栈顶的优先级
                q.push(s.top());//将栈顶的操作符不断弹出，直到栈顶优先级小于读入操作符
                s.pop();
            }
            temp.op=str[i];//将读入的操作符放入栈中
            s.push(temp);
            i++;
        }


    }
    while(!s.empty()){//读完表达式值后，如果栈不为空，则将所有的元素弹出进后缀表达式队列
        q.push(s.top());
        s.pop();

    }
}
double Cal(){
    double temp1,temp2;
    node cur,temp;
    while(!q.empty()){//队列不为空依次出队列
        cur=q.front();
        q.pop();
        if(cur.flag==true)s.push(cur);//若出队元素为操作数，压入栈中
        else{//若出队元素为操作符，则取出栈顶两个元素，进行运算
            temp2=s.top().num;//注意 先出栈的为第二个操作数
            s.pop();
            temp1=s.top().num;
            s.pop();
            temp.flag=true;
            if(cur.op=='+')temp.num=temp1+temp2;
            else if(cur.op=='*')temp.num=temp1*temp2;
            else if(cur.op=='-')temp.num=temp1-temp2;
            else temp.num=temp1/temp2;
            s.push(temp);//计算完后，再次压入栈中
        }
    }
    return s.top().num;

}
int main(){
    op['+']=op['-']=1;//设置操作符优先级
    op['*']=op['/']=2;
    while(getline(cin,str),str!="0"){
        for(string::iterator it=str.end();it!=str.begin();it--){
            if(*it==' ')str.erase(it);//去除空格
        }
        while(!s.empty())s.pop();
        Change();
        printf("%.2f\n",Cal());

    }return 0;
}
