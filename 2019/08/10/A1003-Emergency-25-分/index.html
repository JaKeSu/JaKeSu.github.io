1003 Emergency (25 分)

As an emergency rescue team leader of a city, you are given a special map of your country. The map shows several scattered cities connected by some roads. Amount of rescue teams in each city and the length of each road between any pair of cities are marked on the map. When there is an emergency call to you from some other city, your job is to lead your men to the place as quickly as possible, and at the mean time, call up as many hands on the way as possible.

Input Specification:

Each input file contains one test case. For each test case, the first line contains 4 positive integers: N (≤500) - the number of cities (and the cities are numbered from0to N−1), M- the number of roads, C1 and C2 - the cities that you are currently in and that you must save, respectively. The next line contains N integers, where the i-th integer is the number of rescue teams in the i-th city. Then M lines follow, each describes a road with three integers c1 , c2 and L, which are the pair of cities connected by a road and the length of that road, respectively. It is guaranteed that there exists at least one path from C1 to C2.

Output Specification:

For each test case, print in one line two numbers: the number of different shortest paths between C1 and C2 , and the maximum amount of rescue teams you can possibly gather. All the numbers in a line must be separated by exactly one space, and there is no extra space allowed at the end of a line.

Sample Input:

5 6 0 2
1 2 1 5 3
0 1 1
0 2 2
0 3 1
1 2 1
2 4 1
3 4 1

Sample Output:

2 4

题意

你作为一个城市的救援队领导者，知道不同城市的连接线路的长度，每个城市的救援队个数，现在让你前往目的地，要求路程最短。
输入格式：每个样例，第一行有4个正数，N代表城市的个数，M代表路径条数，c1 , c2表示你当前的城市以及目的城市的标号。
接下来的一行包含N个数字，为每个城市的救援队个数。然后的M行，每一行有三个数，c1 , c2 和L，分别为两个城市的标号以及距离。
输出格式：每个测试样例打印两个数字，最短路径的条数和你能够调动的救援队的最大值，也就是最大边权。

思路

主要思路还是dijkstra算法。由于要打印最短路径条数，用一个num[1]表示节点i到s的最短路径条数，我们在每次通过点v到i找到另外一条最短路径的时候，都通过num[v]更新这个num[i]。而在最短路径相同时，我们要输出点权最大的，于是也用一个w[i]来表示i的最大点权，和num[i]一样，每次找到相同的最短路径时，我们进行比较，如果点权更大则更新。
具体的在代码注释里面写得比较详细。
采用三种方案，进行求解。

代码

1.Dijkstra

#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int MAXV=510;//最大定点数
const int INF=1000000000;//不可达

int G[MAXV][MAXV],d[MAXV],n,w[MAXV],num[MAXV],weight[MAXV],m,s,e;
//定义图G   d数组用于存储最短路径  num数组存储最短路径条数  weight数组存储点权 
bool vis[MAXV]={false};//判断是否已经访问过的顶点
void Dijkstra(int s){
    fill(d,d+MAXV,INF);//初始化最短路径均为无穷大
    memset(num,0,sizeof(num));
    memset(w,0,sizeof(w));//w存储最优的救援队个数
    w[s]=weight[s];//初始化s点的w即为本身的救援队个数，最短路径初始化为一条，s点的最短路径为0
    num[s]=1;
    d[s]=0;
    for(int i=0;i<n;i++){
        int u=-1,MIN=INF;//找出一个点u未被访问，并且距离s点最短
        for(int j=0;j<n;j++){
            if(vis[j]==false&&d[j]<MIN){
                u=j;
                MIN=d[j];
            }
        }
        if(u==-1)return ;
        vis[u]=true;//设置点u已经被访问
        for(int v=0;v<n;v++){
            if(vis[v]==false&&G[u][v]!=INF){//寻找点u附近的可达点，并且没有访问过的点v
                if(d[u]+G[u][v]<d[v]){//如果通过点u可以使得v到s的距离更短，那么优化他
                    d[v]=d[u]+G[u][v];
                    w[v]=w[u]+weight[v];//因为第一次找到最短路径，那么我们将这个点的最优点权更新
                    num[v]=num[u];//v点继承u的最短路径条数，因为u到s有多少条最短路径，那我v经过u到s也应该有这么多条
                }else if(d[u]+G[u][v]==d[v]){//在出现两个最短路径的时候，判断如果这条路可以使得点权w更大（也就是救援队人数增多）我们就更新点权
                    if(w[u]+weight[v]>w[v]){
                        w[v]=w[u]+weight[v];
                    }
                    num[v]+=num[u];//此时由于出现了多条最短路径，所以我们应该让v本来的条数加上u的条数
                }
            }
        }
    }

}
int main(){
    scanf("%d%d%d%d",&n,&m,&s,&e);
    for(int i=0;i<n;i++){
        scanf("%d",&weight[i]);
    }
    int u,v;
    fill(G[0],G[0]+MAXV*MAXV,INF);
    for(int i=0;i<m;i++){
        scanf("%d%d",&u,&v);
        scanf("%d",&G[u][v]);
        G[v][u]=G[u][v];
    }
    Dijkstra(s);
    printf("%d %d\n",num[e],w[e]);
    return 0;
}

2.Dijkstra+DFS

#include<cstdio>
#include<vector>
using namespace std;
const int INF=1000000000;
const int MAXV=510;
int n,m,G[MAXV][MAXV],d[MAXV],s,e,weight[MAXV];
bool vis[MAXV]={false};//判断是否访问过这个顶点
vector<int>pre[MAXV];//用一个数组保存当前顶点在路径中的前驱节点
void Dijkstra(int s){
    fill(d,d+MAXV,INF);
    d[s]=0;//初始化最短距离为0
    for(int i=0;i<n;i++){
        int u=-1,MIN=INF;
        for(int j=0;j<n;j++){//找到距离当前节点可到达的最短距离的节点
            if(vis[j]==false&&d[j]<MIN){
                u=j;
                MIN=d[j];
            }
        }
        if(u==-1)return;
        vis[u]=true;//设置这个节点为已经访问过
        for(int v=0;v<n;v++){//以这个节点为中介，更新所有的最短路径，并且更新前驱节点
            if(vis[v]==false&&G[u][v]!=INF){
                if(d[u]+G[u][v]<d[v]){//如果找到最优点，清除之前此顶点的前驱节点，并加入新的前驱节点
                    d[v]=d[u]+G[u][v];
                    pre[v].clear();
                    pre[v].push_back(u);
                }else if(d[u]+G[u][v]==d[v]){//如果新最优点的最短路径和之前最短距离相等，则将这个新的前驱节点加入进pre数组
                    pre[v].push_back(u);//（这里不用清空，因为是并列关系），说明他们都是符合题意的路径节点
                }
            }
        }
    }
}
int optValue=0,num=0;//定义一个最优值和存储最优路径的数目
vector<int> path,tempPath;
void DFS(int v){
    if(v==s){
        num++;//最优路径数目在每次DFS到叶子节点时+1
        tempPath.push_back(v);
        // int value1=0;
        // for(int i=tempPath.size()-1;i>0;i--){
        //     int id=tempPath[i],idNext=tempPath[i-1];
        //     value1+=G[id][idNext];
        // }
        int value2=0;
        for(int i=tempPath.size()-1;i>=0;i--){//这里value为计算当前路径的点权，也就是救援队个数
            int id=tempPath[i];
            value2+=weight[id];
        }
        if(value2>optValue){//每条路径都要计算他的value，和最优的value进行对比
            optValue=value2;
            path=tempPath;//存储最优路径
        }
        tempPath.pop_back();//删除加入节点，防止污染下一条路径
        return ;
    }
    tempPath.push_back(v);
    for(int i=0;i<pre[v].size();i++){//对前驱节点数组进行遍历，找到所有的路径
        DFS(pre[v][i]);
    }
    tempPath.pop_back();
}
int main(){
    scanf("%d%d%d%d",&n,&m,&s,&e);
    for(int i=0;i<n;i++){
        scanf("%d",&weight[i]);//输入点权，即救援队个数
    }
    int v,u;
    fill(G[0],G[0]+MAXV*MAXV,INF);//初始化图边权
    for(int i=0;i<m;i++){//输入边权，即两个城市之间距离
        scanf("%d%d",&v,&u);
        scanf("%d",&G[v][u]);
        G[u][v]=G[v][u];
    }
    Dijkstra(s);//从起点形成所有的最短距离
    DFS(e);//从终点递归找到最优值
    printf("%d %d\n",num,optValue);
    return 0;
}

3.Bellman-Ford

#include<cstdio>
#include<vector>
#include<set>
#include<cstring>
using namespace std;
const int MAXV=510;
const int INF=1000000000;
struct Node{
    int v,dis;
    Node(int _v,int _dis):v(_v),dis(_dis){}
};
vector<Node> Adj[MAXV];
int n,m,s,e,weight[MAXV],d[MAXV],w[MAXV],num[MAXV];
set<int>pre[MAXV];
void Bellman(int s){
    fill(d,d+MAXV,INF);
    memset(w,0,sizeof(w));
    memset(num,0,sizeof(num));
    d[s]=0;
    w[s]=weight[s];
    num[s]=1;
    for(int i=0;i<n-1;i++){
        for(int u=0;u<n;u++){
            for(int j=0;j<Adj[u].size();j++){
                int v=Adj[u][j].v;
                int dis=Adj[u][j].dis;
                if(d[u]+dis<d[v]){
                    d[v]=d[u]+dis;
                    w[v]=w[u]+weight[v];
                    num[v]=num[u];
                    pre[v].clear();
                    pre[v].insert(u);

                }else if(d[u]+dis==d[v]){
                    if(w[u]+weight[v]>w[v]){
                        w[v]=w[u]+weight[v];
                    }
                    pre[v].insert(u);

                    num[v]=0;
                    set<int>::iterator it;
                    for( it=pre[v].begin();it!=pre[v].end();it++){
                        num[v]+=num[*it];
                    }
                }
            }
        }
    }
}
int main(){
    scanf("%d%d%d%d",&n,&m,&s,&e);
    for( int i=0;i<n;i++){
        scanf("%d",&weight[i]);
    }
    int u,v,wt;
    for(int i=0;i<m;i++){
        scanf("%d%d%d",&u,&v,&wt);
        Adj[u].push_back(Node(v,wt));
        Adj[v].push_back(Node(u,wt));
    }
    Bellman(s);
    printf("%d %d\n",num[e],w[e]);
    return 0;
}
