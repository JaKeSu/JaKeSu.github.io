1057 Stack (30 分)

Stack is one of the most fundamental data structures, which is based on the principle of Last In First Out (LIFO). The basic operations include Push (inserting an element onto the top position) and Pop (deleting the top element). Now you are supposed to implement a stack with an extra operation: PeekMedian – return the median value of all the elements in the stack. With N elements, the median value is defined to be the (N/2)-th smallest element if N is even, or ((N+1)/2)-th if N is odd.

Input Specification:

Each input file contains one test case. For each case, the first line contains a positive integer N (≤105 ). ThenNlines follow, each contains a command in one of the following 3 formats: 

Push key
Pop
PeekMedian

where key is a positive integer no more than 105 .

Output Specification:

For each Push command, insert key into the stack and output nothing. For each Pop or PeekMedian command, print in a line the corresponding returned value. If the command is invalid, print Invalid instead.

Sample Input:

Pop
PeekMedian
Push 3
PeekMedian
Push 2
PeekMedian
Push 1
PeekMedian
Pop
Pop
Push 5
Push 4
PeekMedian
Pop
Pop
Pop
Pop

题意

要求你在栈原有的基础上，实现求栈内元素中位数的操作。

思路

如果简单的每次求中位数，都对栈内元素进行一次遍历然后计算，肯定会超时，因为题目是105级别的。
参考晴神和柳神的代码，主要有两种思想，一是采用分块思想，一种是采用树状数组思想。

分块：

分块思想有点类似hash表的做法，用一个table数组存储入栈的数字个数，比如table[i]存储i元素的个数，每入栈一个元素i，我们就table[i]++，而为了查找速度的快速，我们用一个小一点的分块数组block[]来存储一个区间内，栈内数字出现的个数，例如将整个table分成√n块（向上取整），然后每一块记录√n个元素中，刚刚存进去的元素个数。如果前x个block的和sum小于要找的中位数k，而前x+1个block的和大于k，说明中位数就在第block[x+1]中，通过计算在table中找到这个区间，然后在这个区间将table[y]一个一个加进sum，知道刚好等于k，则y的值即为中位数。

树状数组：

树状数组可以直接解决前缀和的问题，我们用一个树状数组来存储入栈元素的个数，然后利用二分法查找前缀和getSum，就可以求出这个中位数的值。

代码

分块 

#include<cstdio>
#include<stack>
#include<cstring>
using namespace std;
const int maxn =100010;
const int sqrN=316;
stack<int> st;
int block[sqrN];
int table[maxn];
void peekMedian(int k){
    int sum=0;
    int idx=0;
    while(sum+block[idx]<k){
        sum+=block[idx++];
    }
    int num=idx*sqrN;
    while(sum+table[num]<k){
        sum+=table[num++];
    }
    printf("%d\n",num);
}
void Push(int x){
    st.push(x);
    block[x/sqrN]++;
    table[x]++;
}
void Pop(){
    int x=st.top();
    st.pop();
    block[x/sqrN]--;
    table[x]--;
    printf("%d\n",x);
}
int main(){
    int n,data;
    char cmd[15];
    memset(block,0,sizeof(block));
    memset(table,0,sizeof(table));
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%s",cmd);
        if(cmd[1]=='o'){
            if(st.empty()){
                printf("Invalid\n");
            }else 
            Pop();
        }else if(cmd[1]=='u'){
            scanf("%d",&data);
            Push(data);
        }else{
            if(st.empty()){
                printf("Invalid\n");
            }else{
                int k=(st.size()+1)/2;
                peekMedian(k);
            }
        }
    }
}

树状数组 

#include<cstdio>
#include<stack>
using namespace std;
#define lowbit(i) ((i)&(-i))//lowbit()运算
const int maxn=100010;
stack <int> s;
int c[maxn];
void update(int x,int v){//更新树状数组
    for(int i=x;i<maxn;i+=lowbit(i)){
        c[i]+=v;
    }
}
int getSum(int x){//求前x个元素的和
    int sum=0;
    for(int i=x;i>0;i-=lowbit(i)){
        sum+=c[i];
    }
    return sum;
}
int peekMedian(){//二分法求中位数
    int left=1,right=maxn,mid,k=(s.size()+1)/2;
    while(left<right){
        mid=(left+right)/2;
        if(getSum(mid)>=k){
            right=mid;
        }else left=mid+1;
    }
    printf("%d\n",left);
}
int main(){
    int n,data;
    scanf("%d",&n);
    char cmd[10];
    for(int i=0;i<n;i++){
        scanf("%s",cmd);
        if(cmd[1]=='u'){
            scanf("%d",&data);
            s.push(data);
            update(data,1);
        }else if(cmd[1]=='o'){
            if(!s.empty()){
                update(s.top(),-1);
                printf("%d\n",s.top());
                s.pop();
            }else{
                printf("Invalid\n");
            }
        }else {
            if((!s.empty())){
                peekMedian();
            }else printf("Invalid\n");
        }
    }
    return 0;
}
